Paso 1 — Crear la hoja
Necesitas una hoja de Google Sheets en blanco donde se guardarán todos los datos.
📊 Crear hoja nueva en Google Sheets
✅ La hoja se abrirá en una nueva pestaña.
✅ Ponle un nombre descriptivo (ej: FacturaFácil Datos).
✅ No necesitas crear columnas — el sistema las crea solo.
Paso 2 — Abrir Apps Script
Dentro de tu hoja de Google Sheets, abre el editor de scripts.
1
En la hoja, haz clic en el menú Extensiones (barra superior)
2
Selecciona Apps Script
3
Se abre una nueva ventana con el editor de código
4
Borra todo el código que aparece y pega el código del siguiente paso
⚠️ Es importante borrar el código existente (function myFunction(){}) antes de pegar el nuevo.
Paso 3 — Pegar el código
Copia este código y pégalo en el editor de Apps Script.
// ============================================= // FacturaFácil — Google Apps Script v4.0 // DATOS COMPARTIDOS ENTRE DISPOSITIVOS // Pega este código COMPLETO y reemplaza todo // ============================================= // Nombre de la hoja donde se guarda el DB completo var HOJA_DB = "_FacturaFacilDB"; // ── GET: Devuelve todos los datos (para cargar al abrir la app) ──────────── function doGet(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sh = ss.getSheetByName(HOJA_DB); if (!sh || sh.getLastRow() < 2) { return resp({ ok: true, data: null, msg: "Sin datos guardados aún" }); } var raw = sh.getRange(2, 1).getValue(); return resp({ ok: true, data: raw }); } catch(err) { return resp({ ok: false, msg: err.message }); } } function doPost(e) { try { var raw = e.postData ? e.postData.contents : "{}"; var data = JSON.parse(raw); var ss = SpreadsheetApp.getActiveSpreadsheet(); // ── PING ────────────────────────────────────────────────────────── if (data.accion === "ping") { return resp({ ok: true, msg: "Conexión exitosa ✓ — v4.0" }); } // ── GUARDAR DB COMPLETO (método principal) ──────────────────────── // Guarda todo el JSON de la app para compartir entre dispositivos if (data.accion === "saveDB") { var sh = getSheet(ss, HOJA_DB); sh.clearContents(); sh.getRange(1,1).setValue("timestamp"); sh.getRange(1,2).setValue("data"); sh.getRange(2,1).setValue(new Date().toISOString()); sh.getRange(2,2).setValue(data.db); sh.setColumnWidth(2, 600); return resp({ ok: true, msg: "Datos guardados ✓" }); } // ── SINCRONIZAR HOJAS LEGIBLES (tablas visuales en Sheets) ──────── if (data.accion === "syncAll") { // Primero guarda el DB completo var shDB = getSheet(ss, HOJA_DB); shDB.clearContents(); shDB.getRange(1,1).setValue("timestamp"); shDB.getRange(1,2).setValue("data"); shDB.getRange(2,1).setValue(new Date().toISOString()); shDB.getRange(2,2).setValue(data.dbJson || "{}"); // Luego escribe las hojas legibles var hojas = data.hojas; for (var nombre in hojas) { var info = hojas[nombre]; var sh = getSheet(ss, nombre); sh.clearContents(); var rows = []; if (info.headers) rows.push(info.headers); for (var i = 0; i < info.filas.length; i++) { rows.push(info.filas[i].map(sanitize)); } if (rows.length > 0) { sh.getRange(1, 1, rows.length, rows[0].length).setValues(rows); } if (info.headers) { sh.getRange(1,1,1,info.headers.length) .setFontWeight("bold").setBackground("#1D9E75").setFontColor("white"); } } return resp({ ok: true, msg: "Sincronización completa ✓" }); } // ── AGREGAR UNA FILA (sync en tiempo real) ──────────────────────── if (data.accion === "appendRow") { var sh = getSheet(ss, data.hoja); if (sh.getLastRow() === 0 && data.headers) sh.appendRow(data.headers); sh.appendRow(data.fila.map(sanitize)); // También actualizar el DB completo si viene incluido if (data.dbJson) { var shDB2 = getSheet(ss, HOJA_DB); if (shDB2.getLastRow() >= 2) { shDB2.getRange(2,1).setValue(new Date().toISOString()); shDB2.getRange(2,2).setValue(data.dbJson); } } return resp({ ok: true }); } return resp({ ok: false, msg: "Acción no reconocida: " + data.accion }); } catch (err) { return resp({ ok: false, msg: "Error: " + err.message }); } } // ── Helpers ─────────────────────────────────────────────────────────────── function getSheet(ss, nombre) { return ss.getSheetByName(nombre) || ss.insertSheet(nombre); } function sanitize(v) { if (v === null || v === undefined) return ""; if (typeof v === "string") return v.substring(0, 50000); return v; } function resp(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
Después de pegar el código:
1. Guarda con Ctrl + S (o ⌘+S en Mac)
2. Haz clic en Implementar → Nueva implementación
3. Tipo: Aplicación web
4. Ejecutar como: Yo mismo (tu cuenta)
5. Quién tiene acceso: Cualquier persona
6. Clic en Implementar → autoriza permisos
7. Copia la URL del Web App que aparece
🔴 Importante: Si modificas el script después, debes crear una nueva implementación (no "Gestionar implementaciones"). La URL cambia cada vez.
Paso 4 — Pegar la URL y probar
Pega aquí la URL del Web App que copiaste de Google Apps Script.
☁️ Cómo funciona entre dispositivos:
Cada vez que abras la app en cualquier celular, tablet o laptop, se cargará automáticamente la versión más reciente de tus datos desde Google Sheets. Los cambios se guardan en la nube automáticamente.
💡 La URL tiene este formato:
https://script.google.com/macros/s/XXXXXXX/exec

Cómo usar FacturaFácil

1. Configura tu empresa
Ve a Config y llena los datos de tu negocio primero.

2. Carga tus productos
En Inventario → Importar, sube un CSV o escribe los datos directamente.

3. Registra clientes
En Clientes → Nuevo, llena los datos y georreferencia la ubicación.

4. Crea facturas
En Facturar, selecciona cliente, agrega productos y presiona Imprimir.

5. Sincroniza con Google Sheets
En Config → Google Sheets, sigue las instrucciones para conectar tu hoja.
